4}Image slideshow
==============================================================
<!DOCTYPE html>
<html>
<head>
  <title>Image Slideshow</title>
</head>
<body align="center">

<h3>Image Slideshow</h3>
<img id="slide" src="image1.jpg" width="300" height="200"><br>

<button onclick="prev()">Previous</button>
<button onclick="next()">Next</button>

<script>
  let images = ["image1.jpg", "image2.jpg", "image3.jpg"];
  let index = 0;

  function show()
   {
    document.getElementById("slide").src = images[index];
  }

  function next() 
  {
    index = (index + 1) % images.length;
    show();
  }

  function prev()
   {
    index = (index - 1 + images.length) % images.length;
    show();
  }

  setInterval(next, 3000); // Auto slide every 3 seconds
</script>

</body>
</html>

8}Basic calculator
==============================================================
<!DOCTYPE html>
<html>
<head><title>Basic Calculator</title></head>
<body align="center">
<h3>Simple Calculator</h3>
<input id="num1" type="number">
<input id="num2" type="number"><br><br>
<button onclick="calc('+')">Add</button>
<button onclick="calc('-')">Subtract</button>
<button onclick="calc('*')">Multiply</button>
<button onclick="calc('/')">Divide</button>
<p id="result"></p>
<script>
function calc(op){
 let a=parseFloat(document.getElementById("num1").value);
 let b=parseFloat(document.getElementById("num2").value);
 if(isNaN(a)||isNaN(b)){document.getElementById("result").innerText="Error";return;}
 let res=0;
 if(op==='+')res=a+b;
 else if(op==='-')res=a-b;
 else if(op==='*')res=a*b;
 else if(op==='/')res=a/b;
 document.getElementById("result").innerText="Result: "+res;
}
</script>
</body></html>
